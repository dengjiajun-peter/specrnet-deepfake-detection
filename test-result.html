class NoisyLFCCDataset(Dataset):
    def __init__(self, metadata_csv, feature_dir, split, snr_db=None):
        df = pd.read_csv(metadata_csv)
        self.df = df[df['split'] == split].reset_index(drop=True)
        self.feature_dir = Path(feature_dir)
        self.snr_db = snr_db # 信噪比 (dB)，None 表示不加噪

    def __len__(self):
        return len(self.df)

    def add_noise(self, feat, snr_db):
        # 计算信号功率
        sig_power = np.mean(feat**2)
        # 根据 SNR 计算噪声功率
        noise_power = sig_power / (10**(snr_db / 10))
        # 生成高斯噪声
        noise = np.random.normal(0, np.sqrt(noise_power), feat.shape)
        return feat + noise

    def __getitem__(self, idx):
        row = self.df.iloc[idx]
        feat_path = self.feature_dir / f"{row['label']}_{row['utt_id']}.npy"
        feat = np.load(feat_path)
        
        # 添加噪声 (在标准化之前)
        if self.snr_db is not None:
            feat = self.add_noise(feat, self.snr_db)
            
        # Z-Score 标准化 (必须在加噪后重新计算)
        feat = (feat - feat.mean()) / (feat.std() + 1e-6)
        
        return torch.from_numpy(feat).unsqueeze(0).float(), torch.tensor(row['label_id'])

# 1. 定义 LFCC 提取逻辑
def extract_lfcc_custom(y, sr=16000, n_lfcc=80):
    stft = np.abs(librosa.stft(y, n_fft=512, hop_length=160, win_length=400))
    freqs = np.linspace(0, sr / 2, 512 // 2 + 1)
    edges = np.linspace(0, sr / 2, n_lfcc + 2)
    fb = np.zeros((n_lfcc, len(freqs)))
    for i in range(1, n_lfcc + 1):
        l, c, r = edges[i-1], edges[i], edges[i+1]
        fb[i-1] = np.maximum(0, np.minimum((freqs-l)/(c-l), (r-freqs)/(r-c)))
    energies = np.log(np.dot(fb, stft) + 1e-6)
    return dct(energies, type=2, axis=0, norm='ortho')[:n_lfcc].astype(np.float32)
